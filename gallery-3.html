BlogToc( document.getElementById('content') ).build({
  binding: {
    onAfterDataChange: function() {
      $('#content').find('.blogtoc_authorthumbnail a').each(function(){
        if ( !$(this).find('span').length ) {
          $(this).prepend('<span class="icon-user"> </span>');
        }
      });
    },
    onLiveDataChange: function() {
      $('#content').find('.blogtoc_authorthumbnail a').each(function(){
        if ( !$(this).find('span').length ) {
          $(this).prepend('<span class="icon-user"> </span>');
        }
      });
    }
  },
  display: {
    setup: 20
  },
  extendClass: {
    blogtoc_id: "gallery",
    blogtoc_table: "clearfix"
  },
  feed: {
    limit: 50
  },
  progress: {
    render: function(elem, progress) {
      if( elem.className.lastIndexOf('progress') === -1 ) { 
        elem.className += ' progress progress-striped active'; 
      } 

      if( ( div = elem.firstChild ) ) { 
        div.style.width = progress + '%'; 
      } else { 
        div = document.createElement('div'); 
        div.className = 'bar'; 
        div.style.width = progress + '%'; 
        elem.appendChild(div); 
      }
    }
  },
  table: {
    order: ['thumbnail','title','author','comment'],
    showHeader: false
  },
  thumbnail: {
    size: 300,
    authorThumbnail: false
  },
  url: 'portraitsofpainters.blogspot.com'
});
