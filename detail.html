BlogToc HTML Markup

Illustration markup:

[ ... ] => contain the value of the BlogToc options. e.g. if you specified { blogtocId: 'my_blogtoc' }, the 'my_blogtoc' will be applied to [blogtocId]

if( ... ) ... endif => conditional statement.

for( ... ) ... endfor => looping statement.

->loops => The above html markup maybe rendered more than one

<!-- ... --> => The markup maybe appear or not

Use Firebug or Chrome dev tools to inspect the html (Recommended)


When BlogToc is built (with or without user's option), it will generate some HTML markup as shown below

Creating Theme Bundle

Theme bundle is javascript file contains pre-configurations setting for theming BlogToc

Basic markup
(function(){
  
  var loadTheme = function( BlogToc ) {
    (function(){

      BlogToc.theme( 'theme_name', 'path/or/url/to/file.css', {
        // the extendClass option shorthand
      });

    })();
  };

  if ( typeof window !== 'undefined' && window.BlogToc ) {
    loadTheme( window.BlogToc );
  }

})();
Example :
// BlogToc theme configuration
// theme : gotham, @link http://www.gothamcity.com/
// author : Bruce Wayne
// filename : gotham.js

(function(){
  
  var loadTheme = function( BlogToc ) {
    (function(){

      BlogToc.theme( 'gotham', 'css/gotham.css', {
        "id": "gotham", // the blogtoc_id will have class name "gotham"
        "header": "bruce" // the blogtoc_header will have class name "bruce", etc...
        "table": "alfred",
        "footer": "batman",
        "pagination": "joker"
        // The all options in extendClass options can goes here, 
        // make sure you declare it without using "blogtoc_" keyword, because it's already shorthanded.
      });

    })();
  };

  if ( typeof window !== 'undefined' && window.BlogToc ) {
    loadTheme( window.BlogToc );
  }

})();

And then you save the file as gotham.js, and don't forget to also write your CSS rules in css/gotham.css

Example to use it :
// load your theme bundle first
BlogToc.addJS('gotham.js');

BlogToc(element).build({
  theme: {
    setup: "gotham" // must be a theme name in your theme bundle
  }
});

The next step is you might consider to add your theme bundle to our github repository ;)
