$(document).ready(function(){
  
  // Use BlogToc with other libraries (jQuery, momentjs, font-awesome, bootstrap-select)
  BlogToc.addJS('theme/bt_todc-bootstrap.js');

  var myDiv = $("#content")[0];

  BlogToc(myDiv).build({
    binding: {
      onLoaded: function() {
        onBlogtocLoaded();
      },
      onAfterDataChange: function() {
        onAfterBlogtocDataChanged();
        movePostLabelNextToPostTitle();
      },
      onLiveDataChange: function() {
        movePostLabelNextToPostTitle();
      }
    },
    date: {
      render: function(date, option) {
        return moment(date.getDate() + '-' + ( date.getMonth() + 1 ) + '-' + date.getFullYear(), "D-MM-YYYY")
          .format("DD/MM/YYYY");
      }
    },
    display: {
      setup: 25
    },
    extendClass: {
      blogtoc_id: "blogger"
    },
    label: {
      setup: 'All labels'
    },
    language: {
      custom: {
        display: " ",
        search: " ",
        prevPage: "<i class=\"icon-arrow-left-7\"></i>",
        nextPage: "<i class=\"icon-arrow-right-7\"></i>",
        result: "{begin}-{end} of {total}"
      }
    },
    newBadge: {
      setup: 0
    },
    pagination: {
      showFirstPage: false,
      showLastPage: false
    },
    postLabel: {
      separator: ', '
    },
    progress: {
      render: progressRender
    },
    search: {
      markerRender: function(match) {
        return match;
      }
    },
    sorting: {
      key: 'updateDate',
      order: 'descending'
    },
    summary: {
      wordLimit: 80
    },
    table: {
      order: ['index','title','label','author','comment','updateDate'],
      showHeader: false,
      titleWidthPoint: 34,
      labelWidthPoint: 0,
      commentWidthPoint: 4,
      updateDateWidthPoint: 4
    },
    theme : {
      setup: "todc-bootstrap"
    },
    thumbnail: {
      size: 48,
      authorSize: 24,
      authorThumbnail: false
    },
    url: "android-developers.blogspot.com"
  });

  function onBlogtocLoaded() {
    var $content = $("#content");

    // move necessary element to filter
    $content.find('.blogtoc_filter').append( 
      $content.find('.blogtoc_label').addClass('pull-right'), 
      $content.find('.blogtoc_pagination').css('margin','0 5px'), 
      $content.find('.blogtoc_result').removeClass('pull-left').addClass('pull-right') 
    );
    // add blogtoc wrapper div
    $content.find('.blogtoc_filter').after( $('<div class="blogtoc_wrapper"/>') );
    // move table to blogtoc wrapper div
    // & append camouflage and hide it ( for loading trick )
    $content.find('.blogtoc_wrapper').append( 
      $content.find('.blogtoc_table'),
      $('<div class="camouflage" style="display: none"/>')
    );
    // change display position
    $content.find('.blogtoc_display').removeClass('pull-left').addClass('pull-right');
    // add margin-right to query
    $content.find('.blogtoc_query').css('margin-right','5px');

    // change select appearance as button
    $('select').selectpicker({
      width: 'auto'
    });
  }

  function onAfterBlogtocDataChanged() {
    var $content = $("#content");

    // add little loading trick like in the blogger (:P)
    // every time the data changed by using camouflage class

    // show it first
    $content.find('.camouflage').show();
    // and hide it after 1 seconds
    setTimeout(function(){
      $content.find('.camouflage').hide();
    }, 1000);
  }

  function movePostLabelNextToPostTitle() {
    var $content = $("#content");

    // add bootstrap tooltip
    $content.find('.blogtoc_post').tooltip();
    // move post label next to post title
    $content.find('.blogtoc_post').parent().each(function(){
      var $blogtocPostLabel = $(this).next().find('.blogtoc_post_label');

      if ( $blogtocPostLabel.length ) {
        $(this).append( $blogtocPostLabel );
      }
    });
  }

  function progressRender(elem, proggress) {
    var $elem = $(elem);

    // add alert class to blogtoc_loader if haven't yet
    if ( !$elem.hasClass('alert') ) {
      $elem.addClass('alert');
    }

    // append loading text to blogtoc_loader
    if ( $elem.html() === '' ) {
      $elem.append('Loading...');
    }
  }
});
